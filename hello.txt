pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scmGit(
                    branches: [[name: '*/main']],
                    extensions: [],
                    userRemoteConfigs: [[
                        url: 'https://github.com/harinimanga31/MetroRepo'
                    ]]
                )
            }
        }

        stage('Build') {
            steps {
                bat "mvn clean package"
            }
        }
    }
}
